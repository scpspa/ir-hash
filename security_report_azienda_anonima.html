# === Helper: verbose log ===
vlog() {
    [ "$VERBOSE" -eq 1 ] && echo -e "  ${CYAN}[VERBOSE]${NC} $*" || true
}

# === Helper: nanosecond timer (fallback to seconds if %N unsupported) ===
_timer_ns() {
    local NS
    NS=$(date +%s%N 2>/dev/null)
    if [ "${#NS}" -gt 15 ]; then
        echo "$NS"
    else
        echo "$(( $(date +%s) * 1000000000 ))"
    fi
}

# === Helper: add result to JSON array (uses jq for safe escaping) ===
add_result() {
    local CVE="$1" TARGET="$2" PORT="$3" STATUS="$4" DETAIL="$5"
    local DUR_MS=0
    if [ "$_TEST_START" -gt 0 ] 2>/dev/null; then
        local NOW; NOW=$(_timer_ns)
        DUR_MS=$(( (NOW - _TEST_START) / 1000000 ))
        [ "$DUR_MS" -lt 0 ] && DUR_MS=0
    fi
    RESULTS=$(echo "$RESULTS" | jq --arg cve "$CVE" --arg target "$TARGET"         --arg port "$PORT" --arg status "$STATUS" --arg detail "$DETAIL"         --arg technique "$_CURRENT_TECHNIQUE" --argjson duration_ms "$DUR_MS"         --arg ts "$(date -u +%Y-%m-%dT%H:%M:%SZ)"         '. + [{cve:$cve, target:$target, port:$port, status:$status, detail:$detail, technique:$technique, duration_ms:$duration_ms, timestamp:$ts}]')
}

# === Helper: fingerprint server version ===
fingerprint_server() {
    local TARGET="$1"
    vlog "Fingerprinting $TARGET..."
    local HEADERS
    HEADERS=$(curl -sI --connect-timeout 5 -k "https://$TARGET" 2>/dev/null || curl -sI --connect-timeout 5 "http://$TARGET" 2>/dev/null)
    local SERVER
    SERVER=$(echo "$HEADERS" | grep -i "^Server:" | head -1 | sed 's/^[Ss]erver:[[:space:]]*//' | tr -d '\r')
    if [ -n "$SERVER" ]; then
        echo -e "  ${CYAN}[FINGERPRINT]${NC} $TARGET: $SERVER"
    else
        SERVER="(not disclosed)"
        echo -e "  ${CYAN}[FINGERPRINT]${NC} $TARGET: header not disclosed"
    fi
    FINGERPRINT_JSON=$(echo "$FINGERPRINT_JSON" | jq --arg t "$TARGET" --arg s "$SERVER"         '. + [{target:$t, server:$s}]')
}

_JQ_VER_CACHE=""

_detect_jquery_version() {
    local T="$1"
    [ "$_JQ_VER_TARGET" = "$T" ] && return
    _JQ_VER_TARGET="$T"
    _JQ_VER_CACHE=""
    local UA="Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/10.0.0.100 Safari/537.36"
    # Phase 1: Direct WP asset paths
    for PORT in 80 443; do
        PROTO="http"; [ "$PORT" = "443" ] && PROTO="https"
        for P in "/wp-includes/js/jquery/jquery.min.js" "/wp-includes/js/jquery/jquery.js" "/assets/js/jquery.min.js" "/js/jquery.min.js"; do
            local RESP
            RESP=$(curl -s -A "$UA" --connect-timeout "$TIMEOUT" "${PROTO}://${T}:${PORT}${P}" 2>/dev/null | head -5)
            _JQ_VER_CACHE=$(echo "$RESP" | grep -oP 'jQuery v\K[0-9.]+' | head -1)
            if [ -n "$_JQ_VER_CACHE" ]; then
                vlog "jQuery version detected: $_JQ_VER_CACHE (from ${PROTO}://${T}:${PORT}${P})"
                return
            fi
        done
    done
    # Phase 2: Parse main page HTML for jQuery <script> tags
    for PROTO in https http; do
        local PAGE
        PAGE=$(curl -sL -A "$UA" --connect-timeout "$TIMEOUT" -m 15 "${PROTO}://${T}/" 2>/dev/null | head -200)
        [ -z "$PAGE" ] && continue
        # Look for inline jQuery version string (e.g., jQuery.fn.jquery = "3.6.0")
        _JQ_VER_CACHE=$(echo "$PAGE" | grep -oP 'jQuery v\K[0-9.]+' | head -1)
        [ -n "$_JQ_VER_CACHE" ] && { vlog "jQuery version from inline: $_JQ_VER_CACHE"; return; }
        # Extract jQuery script URLs from <script src="...jquery...">
        local JQ_URLS
        JQ_URLS=$(echo "$PAGE" | grep -oP '(?:src|SRC)=["']\K[^"']*jquery[^"']*' | head -3)
        for JQ_URL in $JQ_URLS; do
            # Handle relative URLs
            case "$JQ_URL" in
                http*) ;;  # absolute, keep as-is
                //*) JQ_URL="${PROTO}:${JQ_URL}" ;;  # protocol-relative
                /*) JQ_URL="${PROTO}://${T}${JQ_URL}" ;;  # root-relative
                *) JQ_URL="${PROTO}://${T}/${JQ_URL}" ;;  # relative
            esac
            local JS_RESP
            JS_RESP=$(curl -s -A "$UA" --connect-timeout "$TIMEOUT" -m 10 "$JQ_URL" 2>/dev/null | head -5)
            _JQ_VER_CACHE=$(echo "$JS_RESP" | grep -oP 'jQuery v\K[0-9.]+' | head -1)
            if [ -n "$_JQ_VER_CACHE" ]; then
                vlog "jQuery version detected: $_JQ_VER_CACHE (from $JQ_URL)"
                return
            fi
            # Also try "jQuery JavaScript Library v" pattern
            _JQ_VER_CACHE=$(echo "$JS_RESP" | grep -oP 'jQuery JavaScript Library v\K[0-9.]+' | head -1)
            if [ -n "$_JQ_VER_CACHE" ]; then
                vlog "jQuery version detected: $_JQ_VER_CACHE (from $JQ_URL)"
                return
            fi
        done
    done
    vlog "jQuery version not detected on $T"
}

# === Dependency installer ===
install_tool() {
    local TOOL="$1"
    echo -e "  ${YELLOW}[INSTALL]${NC} Installing $TOOL..."
    case "$TOOL" in
        curl)
            sudo apt-get update -qq && sudo apt-get install -y -qq curl
            ;;
        nmap)
            sudo apt-get update -qq && sudo apt-get install -y -qq nmap
            ;;
        jq)
            sudo apt-get update -qq && sudo apt-get install -y -qq jq
            ;;
        nuclei)
            # Try apt first, then go install
            if command -v apt-get &>/dev/null; then
                sudo apt-get update -qq && sudo apt-get install -y -qq nuclei 2>/dev/null || true
            fi
            if ! command -v nuclei &>/dev/null; then
                if command -v go &>/dev/null; then
                    echo "    Installing nuclei via go..."
                    go install -v github.com/projectdiscovery/nuclei/v3/cmd/nuclei@latest 2>/dev/null || true
                    # Add GOPATH/bin to PATH if needed
                    export PATH="${PATH}:${HOME}/go/bin"
                else
                    echo -e "  ${YELLOW}[WARN]${NC} Cannot install nuclei: go not found. Install manually:"
                    echo "    go install -v github.com/projectdiscovery/nuclei/v3/cmd/nuclei@latest"
                    return 1
                fi
            fi
            ;;
        *)
            echo -e "  ${YELLOW}[WARN]${NC} Don't know how to install: $TOOL"
            return 1
            ;;
    esac
    if command -v "$TOOL" &>/dev/null; then
        echo -e "  ${GREEN}[OK]${NC} $TOOL installed successfully"
        return 0
    else
        echo -e "  ${RED}[FAIL]${NC} Failed to install $TOOL"
        return 1
    fi
}

# === Ensure dependencies ===
ensure_dependencies() {
    echo "=== Checking Dependencies ==="
    local ALL_OK=1

    # Required tools
    for tool in curl jq; do
        if command -v "$tool" &>/dev/null; then
            echo -e "  ${GREEN}[OK]${NC} $tool ($(command -v $tool))"
        else
            echo -e "  ${YELLOW}[MISSING]${NC} $tool - attempting install..."
            if ! install_tool "$tool"; then
                echo -e "  ${RED}[FATAL]${NC} $tool is required. Aborting."
                exit 1
            fi
        fi
    done

    # Recommended tools
    for tool in nmap nuclei commix sqlmap dalfox; do
        if command -v "$tool" &>/dev/null; then
            echo -e "  ${GREEN}[OK]${NC} $tool ($(command -v $tool))"
        else
            echo -e "  ${YELLOW}[MISSING]${NC} $tool - attempting install..."
            install_tool "$tool" || echo -e "  ${YELLOW}[WARN]${NC} $tool not available, some checks will be skipped"
        fi
    done
    echo ""
}

    done
    echo ""
}

# === Collect dependency versions ===
_collect_dependencies() {
    local DEP_JSON="{}"
    for tool in curl jq nmap nuclei openssl ncat; do
        local ver=""
        case "$tool" in
            curl)    ver=$(curl --version 2>/dev/null | head -1 | awk '{print $2}') ;;
            jq)      ver=$(jq --version 2>/dev/null | sed 's/jq-//') ;;
            nmap)    ver=$(nmap --version 2>/dev/null | grep -oP 'Nmap version \K[0-9.]+' | head -1) ;;
            nuclei)  ver=$(nuclei -version 2>&1 | grep -oP '[0-9]+\.[0-9]+\.[0-9]+' | head -1) ;;
            openssl) ver=$(openssl version 2>/dev/null | awk '{print $2}') ;;
            ncat)    ver=$(ncat --version 2>&1 | grep -oP '[0-9]+\.[0-9]+' | head -1) ;;
        esac
        if [ -n "$ver" ]; then
            DEP_JSON=$(echo "$DEP_JSON" | jq --arg k "$tool" --arg v "$ver" '. + {($k): $v}')
        fi
    done
    echo "$DEP_JSON"
}

    def __init__(self, timeout=10, verbose=False):
        self.timeout = timeout

        self.results = []

        # Enable requests debug logging in verbose mode
        if self.verbose:
            import logging as _logging
            import http.client as _hc
            _hc.HTTPConnection.debuglevel = 1
            _logging.basicConfig()
            _logging.getLogger().setLevel(_logging.DEBUG)
            req_log = _logging.getLogger("requests.packages.urllib3")
            req_log.setLevel(_logging.DEBUG)
            req_log.propagate = True

    def _vlog(self, msg):
        """Print message only in verbose mode."""
        if self.verbose:
            print(f"  [VERBOSE] {msg}")

    def _run_subprocess(self, cmd, timeout=60):
        """Run subprocess with verbose propagation."""
        self._vlog(f"exec: {cmd if isinstance(cmd, str) else ' '.join(cmd)}")
        result = subprocess.run(
            cmd, shell=isinstance(cmd, str),
            capture_output=True, text=True, timeout=timeout,
        )
        if self.verbose and result.stderr:
            for line in result.stderr.strip().split("\n")[:10]:
                self._vlog(f"  stderr: {line}")
        return result

    def _add_result(self, cve, target, port, status, detail, evidence=""):
        """Record a test result."""
        self.results.append({
            "cve": cve,
            "target": target,
            "port": port,
            "status": status,
            "detail": detail,
            "evidence": evidence[:500] if evidence else "",
            "timestamp": datetime.now().isoformat(),
        })
        if self.verbose:
            symbol = {"vulnerable": "!", "not_vulnerable": ".", "needs_verification": "?",
                       "skipped": "S", "error": "E"}.get(status, "?")
            print(f"  [{symbol}] {cve} on {target}:{port or '-'} -> {status}: {detail[:80]}")
