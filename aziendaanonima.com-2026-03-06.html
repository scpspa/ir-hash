bash — subdomain-monitor.sh
#!/bin/bash
# Subdomain monitoring script
DOMAIN="aziendaanonima.com"
LOGFILE="/var/log/subdomain-monitor.log"

# Check for new subdomains
curl -s "https://crt.sh/?q=%25.${DOMAIN}&output=json" | \
jq -r '.[].name_value' | \
sort -u > /tmp/current_subdomains.txt

# Compare with previous scan
if [ -f /tmp/previous_subdomains.txt ]; then
    NEW_SUBDOMAINS=$(comm -13 /tmp/previous_subdomains.txt /tmp/current_subdomains.txt)
    if [ ! -z "$NEW_SUBDOMAINS" ]; then
        echo "$(date): New subdomains found: $NEW_SUBDOMAINS" >> $LOGFILE
    fi
fi

cp /tmp/current_subdomains.txt /tmp/previous_subdomains.txt

nginx — /etc/nginx/sites-available/webmail-subdomain
# Webmail subdomain configuration
server {
    listen 80;
    listen 443 ssl;
    server_name webmail.aziendaanonima.com;
    
    # Security headers
    add_header X-Frame-Options DENY;
    add_header X-Content-Type-Options nosniff;
    add_header X-XSS-Protection "1; mode=block";
    
    # Rate limiting
    limit_req_zone $binary_remote_addr zone=webmail:10m rate=5r/m;
    limit_req zone=webmail burst=10 nodelay;
    
    # Custom error pages
    error_page 404 /custom_404.html;
    error_page 500 502 503 504 /custom_50x.html;
    
    location / {
        # If no webmail service, redirect to main site
        return 301 https://aziendaanonima.com;
        
        # Or if webmail service exists:
        # proxy_pass http://webmail_backend;
        # proxy_set_header Host $host;
    }
}
